accelmax_list = []
accelmin_list = []

validcsvfile = []
for csv in csvlist:
    print("\nReading the CSV file {}".format(csv))
    pathsplit = csv.split('/')
    # print(pathsplit[-1])

        continue

        accelx = r.accelx()
        validcsvfile.append(csv)
        accelmax_list.append(np.max(accelx['Message']))
        accelmin_list.append(np.min(accelx['Message']))
