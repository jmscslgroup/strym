speed_list = []
accelx_list = []
for csv in csvlist:
    print("\nReading the CSV file {}".format(csv))m.
    r = strymread(csvfile=csv, dbcfile=dbcfile)

    if r.success is True:
        if r.burst is True:
            print("Burst Data. Skipping ....")
            continue

        speed = r.speed()
        if speed.shape[0] == 0:
            continue
        accelx = r.accelx()
        if accelx.shape[0] == 0:
            continue

        speed_list.append(speed)
        accelx_list.append(accelx)
